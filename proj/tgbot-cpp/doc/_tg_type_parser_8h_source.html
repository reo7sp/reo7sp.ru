104  std::shared_ptr<T> tryParseJson(JsonToTgTypeFunc<T> parseFunc, const boost::property_tree::ptree& data, const std::string& keyName) const {
105  auto treeItem = data.find(keyName);

113  std::vector<std::shared_ptr<T>> parseJsonAndGetArray(JsonToTgTypeFunc<T> parseFunc, const boost::property_tree::ptree& data, const std::string& keyName) const {
114  std::vector<std::shared_ptr<T>> result;
115  auto treeItem = data.find(keyName);
116  if (treeItem == data.not_found()) {
117  return result;
118  }
119  for (const std::pair<const std::string, boost::property_tree::ptree>& innerTreeItem : treeItem->second) {
120  result.push_back((this->*parseFunc)(innerTreeItem.second));
121  }
122  return result;

126  std::vector<std::vector<std::shared_ptr<T>>> parseJsonAndGet2DArray(JsonToTgTypeFunc<T> parseFunc, const boost::property_tree::ptree& data, const std::string& keyName) const {
127  std::vector<std::vector<std::shared_ptr<T>>> result;
128  auto treeItem = data.find(keyName);
129  if (treeItem == data.not_found()) {
130  return result;
131  }
132  for (const std::pair<const std::string, boost::property_tree::ptree>& innerTreeItem : treeItem->second) {
133  std::vector<std::shared_ptr<T>> innerResult;
134  for (const std::pair<const std::string, boost::property_tree::ptree>& innerInnerTreeItem : innerTreeItem.second) {
135  innerResult.push_back((this->*parseFunc)(innerInnerTreeItem.second));
136  }
137  result.push_back(innerResult);
138  }
139  return result;
140  }

142  template<typename T>
143  std::string parseArray(TgTypeToJsonFunc<T> parseFunc, const std::vector<std::shared_ptr<T>>& objects) const {
144  std::string result;
145  result += '[';
146  for (const std::shared_ptr<T>& item : objects) {
147  result += (this->*parseFunc)(item);
148  result += ',';
149  }
150  result.erase(result.length() - 1);

155  template<typename T>
156  std::string parse2DArray(TgTypeToJsonFunc<T> parseFunc, const std::vector<std::vector<std::shared_ptr<T>>>& objects) const {
157  std::string result;
158  result += '[';
